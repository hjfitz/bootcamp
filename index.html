    JavaScript - variables and maths

    Outcomes:

    To be able to create variables, use maths and, if there's enough time, use basic logic in JavaScript

    Toggle Spoiler 
                  
                // I'm a comment 
                
                /**
                * I'm a comment
                * that spans over multiple line-numbers
                */
                
                // creating a variable that you can change:
                let avariable = 1;
                avariable = 2;
                avariable // 2
                
                // creating a variable that can't be changed
                // good practise when you're not expecting a variable to change to use const
                // less bugs will come about
                const bvariable = 1;
                bvariable = 2;
                // syntax error
                
                // equality
                1 == '1'; // true
                1 == 1; // true
                1 === '1' // false. 3 equals compares the type (string vs number)
                // inequality
                1 != '1'; // false, type is compared
                1 !== '1'; // true
                1 > 2;
                1 < 2;
                1 >= 2;
                1 >= 2;
                
                // we can use these equalities to structure a program
                if (1 === 1) {
                  console.log('1 is equal to one!');
                } else {
                  console.log('1 is NOT equal to one!? This shouldn`t happen');
                  console.log('because we know that if statement is untrue, this will never print');
                }
                
                // a loop, that repeats everything in the braces { in here! }
                // until the condition in the brackets is false/wrong
                while ( 1 === 1) { console.log('I run forever!') }
                while (1 > 2) { console.log('You should not see this!') }
                
                let n = 0;
                while (n > 4) { n = n + 1 }
                
                // a for-loop is far more explicit
                // it lets you do something a given amount of times
                // there are 3 parts - the variable you define, the condition that has to be true for it to run
                // and the part that changes that variable you defined
                for (let i = 0; i < 10; i += 1) {
                  console.log('I\'m incrementing i 10 times! the value of i is: ' + i);
                }
                
                // you can even go through a list (shown off next session)
                let myList = [a,b,c,d];
                console.log(myList[0]); // a
                console.log(myList[3]); // d
                let listLength = myList.length; // 4
                for (let i = 0; i <= listLength) {
                  let currentItem = myList[i];
                  console.log(currentItem); // should print a, b, c, d
                }
                
                // and finally, you can use 'if' to find something in an array 
                let names = ['Harry', 'Sam', 'That Northern One', 'Peter', 'Phil'];
                for (let i = 0; i <= names.length; i++) {
                  if (names[i] === 'That Northern One') {
                    console.log('Found that northern one');
                    console.log('he was at place ' + i + 'in the list');
                  }
                }
